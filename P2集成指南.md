## 1️⃣ Redis 缓存集成

### 对话列表缓存

**文件**: `src/modules/messages/conversations.service.js`

```javascript
const cache = require('../../utils/cache');

exports.listConversations = async (userId) => {
  // 1. 尝试从缓存获取
  const cacheKey = cache.keys.conversations(userId);
  let conversations = await cache.get(cacheKey);

  if (conversations) {
    return conversations;  // 命中缓存
  }

  // 2. 缓存未命中，从数据库读取
  conversations = db.prepare(`
    SELECT id, name, type, last_message, unread_count
    FROM conversations
    WHERE user_id = ?
    ORDER BY updated_at DESC
    LIMIT 20
  `).all(userId);

  // 3. 写入缓存（TTL: 5 分钟）
  await cache.set(cacheKey, conversations, 300);

  return conversations;
};

// 重要: 修改对话时删除缓存
exports.updateConversation = async (userId, conversationId, data) => {
  // 执行更新
  db.prepare(`UPDATE conversations SET ... WHERE id = ?`).run(conversationId);

  // 删除缓存，强制下次查询重新加载
  await cache.del(cache.keys.conversations(userId));
};
```

### 用户信息缓存

**文件**: `src/modules/users/users.service.js`

```javascript
const cache = require('../../utils/cache');

exports.getUser = async (userId) => {
  // 尝试缓存
  const cacheKey = cache.keys.user(userId);
  let user = await cache.get(cacheKey);

  if (user) return user;

  // 数据库查询
  user = db.prepare(`SELECT * FROM users WHERE id = ?`).get(userId);

  // 写入缓存（TTL: 30 分钟）
  await cache.set(cacheKey, user, 1800);

  return user;
};

// 更新用户信息时清除缓存
exports.updateUser = async (userId, data) => {
  db.prepare(`UPDATE users SET ...`).run(userId);
  await cache.del(cache.keys.user(userId));
  
  // 如果用户信息影响对话列表（如头像），也清除对话缓存
  await cache.delPattern(`conv:${userId}:*`);
};
```

### 搜索结果缓存

**文件**: `src/modules/messages/messages.service.js`

```javascript
const cache = require('../../utils/cache');

exports.search = async (userId, query) => {
  // 搜索查询作为缓存键（考虑到查询差异大）
  const cacheKey = cache.keys.search(userId, query);
  let results = await cache.get(cacheKey);

  if (results) return results;

  // 执行搜索
  results = db.prepare(`
    SELECT * FROM messages
    WHERE user_id = ? AND content LIKE ?
    ORDER BY created_at DESC
    LIMIT 20
  `).all(userId, `%${query}%`);

  // 缓存（TTL: 10 分钟）
  await cache.set(cacheKey, results, 600);

  return results;
};

// 新消息到达时清除相关缓存
exports.createMessage = async (conversationId, data) => {
  // 保存消息
  const message = db.prepare(`INSERT INTO messages ...`).run(conversationId);

  // 清除涉及用户的所有搜索缓存（因为新消息可能被搜到）
  const userId = data.senderId;
  await cache.delPattern(`search:${userId}:*`);

  // 清除对话缓存（最后一条消息变了）
  await cache.delPattern(`conv:${userId}:*`);

  return message;
};
```

## 2️⃣ 结构化日志集成

### 基本用法

**文件**: `src/modules/auth/auth.controller.js`

```javascript
const { info, warn, error, logPerformance } = require('../../utils/logger');

exports.login = asyncHandler(async (req, res) => {
  const startTime = Date.now();

  try {
    const { phone, password } = req.body;

    // 记录登录尝试
    info('Login attempt', {
      phone,
      ip: req.ip,
    });

    const user = svc.login(req.body);
    const token = jwt.sign({ id: user.id }, config.jwtSecret);

    // 记录性能
    const duration = Date.now() - startTime;
    logPerformance('login', duration, { phone, success: true });

    // 记录成功登录
    info('User logged in', {
      userId: user.id,
      phone,
    });

    res.cookie(config.cookieName, token);
    res.json({ user, token });
  } catch (err) {
    // 记录错误
    error('Login failed', err, {
      phone: req.body.phone,
      errorType: err.name,
    });

    // 记录登录失败
    warn('Login attempt failed', {
      phone: req.body.phone,
      reason: err.message,
      ip: req.ip,
    });

    res.status(400).json({ error: err.message });
  }
});
```

### 性能监控日志

```javascript
const { logPerformance } = require('../../utils/logger');

exports.listConversations = async (userId) => {
  const start = Date.now();

  // 执行操作
  const conversations = fetchFromDB(userId);

  // 记录性能（若 >1s 自动 warn）
  const duration = Date.now() - start;
  logPerformance('listConversations', duration, {
    userId,
    count: conversations.length,
  });

  return conversations;
};
```
